# Dangerfile - Checks for Created & Modified Files

# Get list of created and modified files
created_files = git.added_files
modified_files = git.modified_files

# Output messages
message("✅ Created Files: #{created_files.join(', ')}") unless created_files.empty?
message("✅ Modified Files: #{modified_files.join(', ')}") unless modified_files.empty?

# Warn if no changes are detected
warn("⚠️ No files have been created or modified!") if created_files.empty? && modified_files.empty?
